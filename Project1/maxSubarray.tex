			\begin{algorithm}[H]
				\SetLine
				\linesnumbered
				\dontprintsemicolon
				\KwData{Integer array A of size N}
				\KwResult{Greatest Sum of Subarray}
				\For{$i \gets 0$ \KwTo $N$}{
					\For{$j \gets i$ \KwTo $N$}{
						$s \gets 0$\;
						\For{$k \gets i$ \KwTo $j$}{
							$s \gets s + A[k]$\;
						}
						\If{$s > max$}{
							 $max \gets s$\;
						}
					}
				}
			\caption{Pseudocode for Basic Enumeration}
			\end{algorithm}
